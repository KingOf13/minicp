package minicp.engine.core;

public class BoolVarImpl extends IntVarImpl implements BoolVar {

    public BoolVarImpl(Solver cp) {
        super(cp, 0, 1);
    }

    @Override
    public boolean isTrue() {
        return min() == 1;
    }

    @Override
    public boolean isFalse() {
        return max() == 0;
    }

    @Override
    public void assign(boolean b) {
        assign(b ? 1 : 0);
    }
}
